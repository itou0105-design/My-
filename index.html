<script>

const COLORS = ['#ff6b35','#ffd23f','#06d6a0','#a855f7','#38bdf8','#f472b6','#fb923c','#34d399'];

let specialPeople = [];

let pidCounter = 0;

function addPerson() {

  const id = ++pidCounter;

  specialPeople.push({ id, name: '上司', ratio: 2.0 });

  renderSpecialList();

}

function removePerson(id) {

  specialPeople = specialPeople.filter(p => p.id !== id);

  renderSpecialList();

}

function renderSpecialList() {

  const container = document.getElementById('special-list');

  container.innerHTML = '';

  specialPeople.forEach((person, idx) => {

    const div = document.createElement('div');

    div.className = 'special-person';

    div.innerHTML = `

      <div class="special-person-top">

        <input class="person-name-input" type="text" value="${person.name}" placeholder="名前"

          oninput="specialPeople[${idx}].name = this.value">

        <button class="remove-btn" onclick="removePerson(${person.id})">✕</button>

      </div>

      <div class="slider-row">

        <span class="slider-label">倍率</span>

        <input type="range" min="1.0" max="5.0" step="0.1" value="${person.ratio}"

          style="accent-color:var(--accent)"

          oninput="specialPeople[${idx}].ratio = parseFloat(this.value); this.nextElementSibling.textContent = parseFloat(this.value).toFixed(1) + 'x'">

        <span class="slider-val">${person.ratio.toFixed(1)}x</span>

      </div>

    `;

    container.appendChild(div);

  });

}

function fmt(n) {

  return '¥' + Math.round(n).toLocaleString('ja-JP');

}

function calculate() {

  const total = parseFloat(document.getElementById('total').value) || 0;

  const people = parseInt(document.getElementById('people').value) || 0;

  if (total <= 0 || people <= 0) {

    alert('合計金額と参加人数を入力してください。');

    return;

  }

  // 有効な特別負担者（人数以内）

  const validSpecial = specialPeople.slice(0, people - 1);

  const normalCount = people - validSpecial.length;

  // 合計「口数」

  const totalUnits = normalCount * 1.0 + validSpecial.reduce((s, p) => s + p.ratio, 0);

  const unitPrice = total / totalUnits;

  const normalAmount = Math.ceil(unitPrice / 10) * 10; // 10円単位切り上げ

  // 特別負担者の金額

  const specialAmounts = validSpecial.map(p => Math.ceil(unitPrice * p.ratio / 10) * 10);

  // 端数

  const collectedTotal = normalCount * normalAmount + specialAmounts.reduce((s, a) => s + a, 0);

  const diff = collectedTotal - total;

  // ── 結果表示 ──

  document.getElementById('res-total').textContent = fmt(total);

  document.getElementById('res-people').textContent = people + ' 人';

  document.getElementById('res-normal').textContent = fmt(normalAmount);

  document.getElementById('res-count').textContent = `一般参加者 ${normalCount} 名`;

  document.getElementById('rounding-note').textContent =

    diff >= 0 ? `※ 端数調整 +${fmt(diff)} の余り（幹事預かりなど）`

              : `※ 端数 ${fmt(Math.abs(diff))} 不足（幹事が調整）`;

  // 人別リスト

  const listEl = document.getElementById('person-list');

  listEl.innerHTML = '';

  const allEntries = [];

  // 一般参加者まとめ

  allEntries.push({ name: `一般参加者（${normalCount}名）`, amount: normalAmount, pct: (normalAmount / total * normalCount), color: COLORS[0], isNormal: true, count: normalCount });

  // 特別負担者

  validSpecial.forEach((p, i) => {

    allEntries.push({ name: p.name, amount: specialAmounts[i], ratio: p.ratio, color: COLORS[i + 1] || COLORS[0], isNormal: false });

  });

  allEntries.forEach(e => {

    const row = document.createElement('div');

    row.className = 'person-result';

    const totalPaid = e.isNormal ? e.amount * e.count : e.amount;

    const pct = (totalPaid / collectedTotal * 100).toFixed(1);

    row.innerHTML = `

      <div class="pr-left">

        <div class="pr-dot" style="background:${e.color}"></div>

        <div>

          <div class="pr-name">${e.name}${!e.isNormal ? ` <span class="pr-tag">${e.ratio.toFixed(1)}x</span>` : ''}</div>

          ${e.isNormal ? `<div style="font-size:11px;color:var(--text-muted)">1人あたり ${fmt(e.amount)}</div>` : ''}

        </div>

      </div>

      <div class="pr-right">

        <div class="pr-amount" style="color:${e.color}">${e.isNormal ? fmt(e.amount) : fmt(e.amount)}</div>

        <div class="pr-pct">全体の ${pct}%${e.isNormal ? ` × ${e.count}人` : ''}</div>

      </div>

    `;

    listEl.appendChild(row);

  });

  // ── 円グラフ ──

  const canvas = document.getElementById('pie-canvas');

  const ctx = canvas.getContext('2d');

  const cx = 90, cy = 90, r = 80, ir = 44;

  // データ

  const pieData = [];

  pieData.push({ label: `一般 (${normalCount}名)`, value: normalAmount * normalCount, color: COLORS[0] });

  validSpecial.forEach((p, i) => {

    pieData.push({ label: p.name, value: specialAmounts[i], color: COLORS[i + 1] || COLORS[0] });

  });

  const pieTotal = pieData.reduce((s, d) => s + d.value, 0);

  ctx.clearRect(0, 0, 180, 180);

  let startAngle = -Math.PI / 2;

  pieData.forEach((d, i) => {

    const slice = (d.value / pieTotal) * Math.PI * 2;

    ctx.beginPath();

    ctx.moveTo(cx, cy);

    ctx.arc(cx, cy, r, startAngle, startAngle + slice);

    ctx.closePath();

    ctx.fillStyle = d.color;

    ctx.fill();

    startAngle += slice;

  });

  // ドーナツ中抜き

  ctx.beginPath();

  ctx.arc(cx, cy, ir, 0, Math.PI * 2);

  ctx.fillStyle = '#1a1928';

  ctx.fill();

  // 中央テキスト

  ctx.fillStyle = '#fffffe';

  ctx.font = 'bold 11px Space Mono, monospace';

  ctx.textAlign = 'center';

  ctx.textBaseline = 'middle';

  ctx.fillText(people + '人', cx, cy - 7);

  ctx.font = '10px Noto Sans JP, sans-serif';

  ctx.fillStyle = '#a7a9be';

  ctx.fillText('参加者', cx, cy + 8);

  // 凡例

  const legendEl = document.getElementById('chart-legend');

  legendEl.innerHTML = '';

  pieData.forEach(d => {

    const pct = (d.value / pieTotal * 100).toFixed(1);

    const item = document.createElement('div');

    item.className = 'legend-item';

    item.innerHTML = `

      <div class="legend-dot" style="background:${d.color}"></div>

      <span style="min-width:90px">${d.label}</span>

      <span class="legend-pct" style="color:${d.color}">${pct}%</span>

    `;

    legendEl.appendChild(item);

  });

  // ── 経費精算メモ ──

  const expenseEl = document.getElementById('expense-rows');

  expenseEl.innerHTML = '';

  const expRows = [];

  expRows.push({ label: `一般参加者 × ${normalCount}名`, value: normalAmount });

  validSpecial.forEach((p, i) => {

    expRows.push({ label: `${p.name}（${p.ratio.toFixed(1)}倍）`, value: specialAmounts[i] });

  });

  expRows.push({ label: '飲食店合計', value: total });

  expRows.forEach((r, i) => {

    const row = document.createElement('div');

    row.className = 'expense-row';

    if (i === expRows.length - 1) {

      row.style.borderTop = '1px solid rgba(6,214,160,0.2)';

      row.style.marginTop = '4px';

      row.style.paddingTop = '8px';

      row.style.fontWeight = '700';

    }

    row.innerHTML = `<span>${r.label}</span><span>${fmt(r.value)}</span>`;

    expenseEl.appendChild(row);

  });

  // 表示

  const sec = document.getElementById('result-section');

  sec.style.display = 'block';

  setTimeout(() => sec.scrollIntoView({ behavior: 'smooth', block: 'start' }), 50);

}

// 初期サンプル

document.getElementById('total').value = 50000;

document.getElementById('people').value = 8;

addPerson();
